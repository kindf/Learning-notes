> > {

> >     {

> >     }
> > };
> > 
> > void ServerAliveMsgBack()	// 服务器心跳包返回
> > {
> >     m_last_avlie_timer = 0;
> > }
> > 
> > void SendAliveMsg()
> > {
> >     // 发送协议
> > }
> > ```
